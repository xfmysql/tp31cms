CREATE TABLE cms_tabinfo(
	`id` int(11)  not null AUTO_INCREMENT,
	`name` VARCHAR(100) NOT NULL,
	`descript`  mediumtext NULL,
	`addtime` datetime NOT NULL,
	 `status` tinyint(1) NOT NULL DEFAULT '1' ,
	`clicks`  INT  NULL,
